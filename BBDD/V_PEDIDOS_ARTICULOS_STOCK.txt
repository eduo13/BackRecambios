SELECT        dbo.PEDIDOS.ID_PEDIDO, dbo.PEDIDOS.ESTADO, dbo.STOCK.CANTIDAD, dbo.PEDIDOS_ARTICULOS.ID_ARTICULO, dbo.PEDIDOS_ARTICULOS.CANTIDAD AS CANTIDADPEDIDA
FROM            dbo.PEDIDOS INNER JOIN
                         dbo.PEDIDOS_ARTICULOS ON dbo.PEDIDOS.ID_PEDIDO = dbo.PEDIDOS_ARTICULOS.ID_PEDIDO INNER JOIN
                         dbo.STOCK ON dbo.PEDIDOS_ARTICULOS.ID_ARTICULO = dbo.STOCK.ID_ARTICULO AND dbo.PEDIDOS_ARTICULOS.CANTIDAD >= dbo.STOCK.CANTIDAD
